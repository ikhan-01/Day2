Manual Queue Implementation

Problem Statement: Implement a queue using a custom class with a Python list.

class Queue:
    def __init__(self):
        self.data = []

    def enqueue(self, val):
        self.data.append(val)

    def dequeue(self):
        if not self.data:
            return "Underflow"
        return self.data.pop(0)

    def is_empty(self):
        return len(self.data) == 0
Time: O(n) for dequeue (due to list shift), O(1) for enqueue

Circular Queue Logic

Problem Statement: Explain how circular queue logic avoids wasted space and implements wrap-around.

class CircularQueue:
    def __init__(self, k):
        self.q = [None]*k
        self.size = k
        self.front = self.rear = -1

    def enqueue(self, val):
        if (self.rear + 1) % self.size == self.front:
            return "Overflow"
        if self.front == -1:
            self.front = 0
        self.rear = (self.rear + 1) % self.size
        self.q[self.rear] = val

    def dequeue(self):
        if self.front == -1:
            return "Underflow"
        temp = self.q[self.front]
        if self.front == self.rear:
            self.front = self.rear = -1
        else:
            self.front = (self.front + 1) % self.size
        return temp
Time: O(1), Space: O(n)

Circular Queue Simulation

Problem Statement: Simulate a circular queue of size 3. Enqueue 1, 2, 3 then dequeue and enqueue 4.

Example:
Input: Enqueue 1, 2, 3 ‚Üí Dequeue ‚Üí Enqueue 4
Final State: [4, 2, 3]
cq = CircularQueue(3)
cq.enqueue(1)
cq.enqueue(2)
cq.enqueue(3)
cq.dequeue()
cq.enqueue(4)
Demonstrates wrap-around after full capacity

Manual Algorithm Using Stack:

def reverse_queue_manual(q):
    stack = []
    while q:
        stack.append(q.pop(0))
    while stack:
        q.append(stack.pop())
def reverse_queue(q):
    if not q:
        return
    x = q.pop(0)
    reverse_queue(q)
    q.append(x)
Time: O(n), Space: O(n) (call stack)

Mock Interview 1

Problem Statement: Implement a queue using two stacks.

Example:
Input: enqueue(1), enqueue(2), dequeue()
Output: 1
class MyQueue:
    def __init__(self):
        self.stack1 = []
        self.stack2 = []

    def enqueue(self, x):
        self.stack1.append(x)

    def dequeue(self):
        if not self.stack2:
            while self.stack1:
                self.stack2.append(self.stack1.pop())
        return self.stack2.pop() if self.stack2 else "Underflow"
Time: Amortized O(1), worst O(n)

Mock Interview 2

Problem Statement: Given a queue, print its elements in reverse order without modifying it permanently.

Example:
Input: [1, 2, 3]
Output: 3 2 1
from collections import deque

def print_reverse(q):
    if not q:
        return
    val = q.popleft()
    print_reverse(q)
    print(val)
    q.appendleft(val)
Time: O(n), Space: O(n) (recursion stack)
