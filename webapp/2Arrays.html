Reverse Array

Problem Statement: Given an array of integers, reverse the elements of the array in-place, i.e., without using extra space.

Brute-force Approach (Python)

def reverse_array_brute(arr):
    result = []
    for i in range(len(arr) - 1, -1, -1):
        result.append(arr[i])
    return result
Time: O(n), Space: O(n) ‚Äì New array created

Optimized Approach (Python)

def reverse_array_optimized(arr):
    start, end = 0, len(arr) - 1
    while start < end:
        arr[start], arr[end] = arr[end], arr[start]
        start += 1
        end -= 1
    return arr
Time: O(n), Space: O(1)

JS Visualization

function reverseArray(arr) {
  let start = 0, end = arr.length - 1;
  while (start < end) {
    console.log(`Swapping ${arr[start]} with ${arr[end]}`);
    [arr[start], arr[end]] = [arr[end], arr[start]];
    start++;
    end--;
  }
  console.log("Result:", arr);
}

Rotate Array

Problem Statement: Given an array, rotate the elements to the right by k steps.

Brute-force Approach (Python)

def rotate_array_brute(arr, k):
    k = k % len(arr)
    for _ in range(k):
        last = arr.pop()
        arr.insert(0, last)
    return arr
Time: O(k*n), Space: O(1)

Optimized Approach (Python)

def rotate_array_optimized(arr, k):
    k = k % len(arr)
    arr[:] = arr[-k:] + arr[:-k]
    return arr
Time: O(n), Space: O(1)

JS Visualization

function rotateArray(arr, k) {
  k %= arr.length;
  let rotated = [...arr.slice(-k), ...arr.slice(0, -k)];
  console.log("Rotated Array:", rotated);
  return rotated;
}

Maximum Element

Problem Statement: Find the maximum value in the array.

Brute-force Approach (Python)

def find_max_brute(arr):
    max_val = arr[0]
    for num in arr:
        if num > max_val:
            max_val = num
    return max_val
Time: O(n), Space: O(1)

Optimized Approach (Python)

def find_max_optimized(arr):
    return max(arr)
Time: O(n), Space: O(1)

JS Visualization

function maxElement(arr) {
  let max = arr[0];
  for (let i = 1; i < arr.length; i++) {
    if (arr[i] > max) max = arr[i];
  }
  console.log("Maximum Element:", max);
  return max;
}

Subarray Sum

Problem Statement: Find the contiguous subarray with the maximum sum.

Brute-force Approach (Python)

def max_subarray_sum_brute(arr):
    max_sum = float('-inf')
    for i in range(len(arr)):
        for j in range(i, len(arr)):
            max_sum = max(max_sum, sum(arr[i:j+1]))
    return max_sum
Time: O(n^2), Space: O(1)

Optimized (Kadane's Algorithm, Python)

def max_subarray_sum_kadane(arr):
    max_sum = curr_sum = arr[0]
    for num in arr[1:]:
        curr_sum = max(num, curr_sum + num)
        max_sum = max(max_sum, curr_sum)
    return max_sum
Time: O(n), Space: O(1)

JS Visualization

function maxSubarraySum(arr) {
  let maxSum = arr[0], current = arr[0];
  for (let i = 1; i < arr.length; i++) {
    current = Math.max(arr[i], current + arr[i]);
    maxSum = Math.max(maxSum, current);
  }
  console.log("Max Subarray Sum:", maxSum);
  return maxSum;
}

Second Largest

Problem Statement: Find the second largest element in the array.

Brute-force Approach (Python)

def second_largest_brute(arr):
    arr = list(set(arr))
    arr.sort()
    return arr[-2]
Time: O(n log n), Space: O(n)

Optimized Approach (Python)

def second_largest_optimized(arr):
    first = second = float('-inf')
    for num in arr:
        if num > first:
            second, first = first, num
        elif num > second and num != first:
            second = num
    return second
Time: O(n), Space: O(1)

JS Visualization

function secondLargest(arr) {
  let first = -Infinity, second = -Infinity;
  for (let num of arr) {
    if (num > first) {
      second = first;
      first = num;
    } else if (num > second && num !== first) {
      second = num;
    }
  }
  console.log("Second Largest:", second);
  return second;
}

Longest Increasing Subarray

Problem Statement: Find the length of the longest contiguous increasing subarray.

def longest_increasing_subarray(arr):
    max_len = cur_len = 1
    for i in range(1, len(arr)):
        if arr[i] > arr[i - 1]:
            cur_len += 1
            max_len = max(max_len, cur_len)
        else:
            cur_len = 1
    return max_len
Time: O(n), Space: O(1)

Pairs with Given Sum

Problem Statement: Find all pairs in the array that sum to a specific target value.

def find_pairs(arr, target):
    seen = set()
    result = []
    for num in arr:
        if target - num in seen:
            result.append((target - num, num))
        seen.add(num)
    return result
