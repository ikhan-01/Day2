Inbuilt Stack Implementation

Problem Statement: Use Python's built-in list methods to simulate stack operations.

stack = []
stack.append(10)  # Push
stack.append(20)
print(stack[-1])  # Peek
stack.pop()       # Pop
Time: O(1) for append, pop, peek | Space: O(n)

Manual Stack Implementation

Problem Statement: Implement stack operations using a custom Stack class backed by a Python list.

class Stack:
    def __init__(self):
        self.data = []

    def push(self, val):
        self.data.append(val)

    def pop(self):
        return self.data.pop() if self.data else "Underflow"

    def peek(self):
        return self.data[-1] if self.data else "Empty"

    def is_empty(self):
        return len(self.data) == 0
Time: O(1), Space: O(n)

Balanced Parentheses

Problem Statement: Check if a string of parentheses is balanced using a stack.

Example:
Input: "(a + b) * {c + [d - e]}"
Output: True (Balanced)
def is_balanced(expr):
    stack = []
    pairs = {')': '(', ']': '[', '}': '{'}
    for ch in expr:
        if ch in pairs.values():
            stack.append(ch)
        elif ch in pairs:
            if not stack or stack.pop() != pairs[ch]:
                return False
    return not stack
Time: O(n), Space: O(n)

Valid Parentheses

Problem Statement: Determine if a string containing '(', ')', '{', '}', '[' and ']' is valid.

Example:
Input: "({[()]})"
Output: True
def is_valid(s):
    stack = []
    mapping = {')': '(', ']': '[', '}': '{'}
    for char in s:
        if char in mapping.values():
            stack.append(char)
        elif char in mapping:
            if not stack or mapping[char] != stack.pop():
                return False
    return not stack
Time: O(n), Space: O(n)

Min Stack

Problem Statement: Design a stack that supports push, pop, top, and retrieving the minimum element in constant time.

Example:
Sequence: push(5), push(2), push(8), getMin()
Output: 2
class MinStack:
    def __init__(self):
        self.stack = []
        self.min_stack = []

    def push(self, x):
        self.stack.append(x)
        if not self.min_stack or x <= self.min_stack[-1]:
            self.min_stack.append(x)

    def pop(self):
        if self.stack.pop() == self.min_stack[-1]:
            self.min_stack.pop()

    def top(self):
        return self.stack[-1]

    def getMin(self):
        return self.min_stack[-1]
Time: O(1), Space: O(n)

Infix to Postfix

Problem Statement: Convert an infix expression to a postfix expression using a stack.

Example:
Input: a+(b*c)
Output: abc*+
def precedence(op):
    if op in ('+', '-'): return 1
    if op in ('*', '/'): return 2
    return 0

def infix_to_postfix(expr):
    stack = []
    result = ''
    for c in expr:
        if c.isalnum():
            result += c
        elif c == '(':
            stack.append(c)
        elif c == ')':
            while stack and stack[-1] != '(':
                result += stack.pop()
            stack.pop()
        else:
            while stack and precedence(c) <= precedence(stack[-1]):
                result += stack.pop()
            stack.append(c)
    while stack:
        result += stack.pop()
    return result
Time: O(n), Space: O(n)

Mock Interview 1

Problem Statement: Implement a stack that supports finding the middle element in O(1) time.

Example:
Sequence: push(1), push(2), push(3), find_middle()
Output: 2
class DLLNode:
    def __init__(self, data):
        self.data = data
        self.prev = None
        self.next = None

class MidStack:
    def __init__(self):
        self.head = None
        self.mid = None
        self.count = 0

    def push(self, val):
        node = DLLNode(val)
        node.next = self.head
        if self.head:
            self.head.prev = node
        self.head = node
        self.count += 1

        if self.count == 1:
            self.mid = node
        elif self.count % 2 == 0:
            self.mid = self.mid.prev

    def pop(self):
        if self.count == 0:
            return "Stack Underflow"
        popped = self.head.data
        self.head = self.head.next
        if self.head:
            self.head.prev = None
        self.count -= 1

        if self.count % 2 == 1:
            self.mid = self.mid.next
        return popped

    def find_middle(self):
        if self.mid:
            return self.mid.data
        return "Empty Stack"
Time: O(1), Space: O(n)

Mock Interview 2

Problem Statement: Evaluate a postfix expression using a stack.

Example:
Input: "5 1 2 + 4 * + 3 -"
Output: 14
def eval_postfix(expr):
    stack = []
    for token in expr.split():
        if token.isdigit():
            stack.append(int(token))
        else:
            b = stack.pop()
            a = stack.pop()
            if token == '+': stack.append(a + b)
            elif token == '-': stack.append(a - b)
            elif token == '*': stack.append(a * b)
            elif token == '/': stack.append(int(a / b))
    return stack.pop()
Time: O(n), Space: O(n)
