Collision Handling

Problem Statement: Explain how hash collisions occur and how chaining solves them.

hash_table = [[] for _ in range(5)]

def insert(key, value):
    index = hash(key) % len(hash_table)
    for pair in hash_table[index]:
        if pair[0] == key:
            pair[1] = value
            return
    hash_table[index].append([key, value])
Time: O(1) avg, O(n) worst | Space: O(n)

Linear Probing

Problem Statement: Handle collisions using linear probing (open addressing).

table_size = 7
keys = [None] * table_size
values = [None] * table_size

def insert(key, value):
    idx = hash(key) % table_size
    while keys[idx] is not None and keys[idx] != key:
        idx = (idx + 1) % table_size
    keys[idx] = key
    values[idx] = value
Time: O(1) avg, O(n) worst | Space: O(n)

Frequency Map

Problem Statement: Count frequency of elements using a hash map.

Example:
Input: [1, 2, 2, 3, 1, 4]
Output: {1: 2, 2: 2, 3: 1, 4: 1}
def frequency_map(arr):
    freq = {}
    for num in arr:
        freq[num] = freq.get(num, 0) + 1
    return freq
Time: O(n), Space: O(n)

Count Pairs with Given Sum

Problem Statement: Count the number of pairs with a given sum using hash map.

Example:
Input: [1, 5, 7, -1, 5], target = 6
Output: 3
def brute_count_pairs(arr, target):
    count = 0
    for i in range(len(arr)):
        for j in range(i + 1, len(arr)):
            if arr[i] + arr[j] == target:
                count += 1
    return count
Time: O(n¬≤), Space: O(1)
def count_pairs(arr, target):
    count = 0
    freq = {}
    for num in arr:
        count += freq.get(target - num, 0)
        freq[num] = freq.get(num, 0) + 1
    return count
Time: O(n), Space: O(n)

Longest Substring Without Repeating Characters

Problem Statement: Find the length of the longest substring with all unique characters.

Example:
Input: "abcabcbb"
Output: 3

Brute Force Approach:

def brute_longest_unique_substring(s):
    max_len = 0
    for i in range(len(s)):
        seen = set()
        for j in range(i, len(s)):
            if s[j] in seen:
                break
            seen.add(s[j])
            max_len = max(max_len, j - i + 1)
    return max_len
Time: O(n¬≤), Space: O(n)
def longest_unique_substring(s):
    seen = {}
    left = max_len = 0
    for right, char in enumerate(s):
        if char in seen and seen[char] >= left:
            left = seen[char] + 1
        seen[char] = right
        max_len = max(max_len, right - left + 1)
    return max_len
Time: O(n), Space: O(n)

Example:

Output: True

Brute Force Approach:

def brute_has_common_element(arr1, arr2):
    for x in arr1:
        for y in arr2:
            if x == y:
                return True
    return False

def has_common_element(arr1, arr2):
    elements = set(arr1)
    for num in arr2:
        if num in elements:
            return True
    return False

Mock Interview 2

Problem Statement: Find the first non-repeating character in a string.

Example:
Input: "aabbcddeff"
Output: c

Brute Force Approach:

def brute_first_unique_char(s):
    for ch in s:
        if s.count(ch) == 1:
            return ch
    return None
Time: O(n¬≤), Space: O(1)
from collections import OrderedDict

def first_unique_char(s):
    freq = OrderedDict()
    for ch in s:
        freq[ch] = freq.get(ch, 0) + 1
    for ch, count in freq.items():
        if count == 1:
            return ch
    return None
Time: O(n), Space: O(n)

Check for Duplicates

Problem Statement: Determine if an array contains any duplicates.

Example:
Input: [1, 2, 3, 4, 1]
Output: True

Brute Force Approach:

def brute_has_duplicates(arr):
    for i in range(len(arr)):
        for j in range(i + 1, len(arr)):
            if arr[i] == arr[j]:
                return True
    return False
Time: O(n¬≤), Space: O(1)
def has_duplicates(arr):
    seen = set()
    for num in arr:
        if num in seen:
            return True
        seen.add(num)
    return False
Time: O(n), Space: O(n)

Subarray with Zero Sum

Problem Statement: Check if there exists a subarray with sum equal to 0.

Example:
Input: [4, 2, -3, 1, 6]
Output: True
def zero_sum_subarray(arr):
    sums = set()
    curr_sum = 0
    for num in arr:
        curr_sum += num
        if curr_sum == 0 or curr_sum in sums:
            return True
        sums.add(curr_sum)
    return False
Time: O(n), Space: O(n)

Count Distinct Elements

Problem Statement: Return the count of distinct elements in the array.

Example:
Input: [1, 2, 2, 3, 4]
Output: 4

Manual Implementation:

def manual_count_distinct(arr):
    seen = []
    for x in arr:
        if x not in seen:
            seen.append(x)
    return len(seen)
Time: O(n¬≤), Space: O(n)
def count_distinct(arr):
    return len(set(arr))
Time: O(n), Space: O(n)

Union of Two Arrays

Problem Statement: Return the union of two arrays.

Example:
Input: [1, 2, 3], [2, 4, 5]
Output: [1, 2, 3, 4, 5]

Manual Implementation:

def manual_union(arr1, arr2):
    result = arr1[:]
    for x in arr2:
        if x not in result:
            result.append(x)
    return result
Time: O(n¬≤), Space: O(n)
def union_arrays(arr1, arr2):
    return list(set(arr1) | set(arr2))
Time: O(n), Space: O(n)

Intersection of Two Arrays

Problem Statement: Return the intersection of two arrays.

Example:
Input: [1, 2, 3], [2, 3, 4]
Output: [2, 3]

Manual Implementation:

def manual_intersection(arr1, arr2):
    result = []
    for x in arr1:
        if x in arr2 and x not in result:
            result.append(x)
    return result
Time: O(n¬≤), Space: O(n)
def intersection_arrays(arr1, arr2):
    return list(set(arr1) & set(arr2))
Time: O(n), Space: O(n)
